static SinglyLinkedListNode insertNodeAtHead(SinglyLinkedListNode llist, int data) {
        
        SinglyLinkedListNode temp = llist;
        SinglyLinkedListNode newNode = new SinglyLinkedListNode(data);
        
        if(llist == null){
            llist = newNode;
            return llist;
        }
        
        while(llist!=null){
            temp = newNode;
            temp.next = llist;
            return temp;
        }
        
        return temp;
        
        
    }
