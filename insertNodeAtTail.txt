    static SinglyLinkedListNode insertNodeAtTail(SinglyLinkedListNode head, int data) {
        
        

SinglyLinkedListNode newNode = new SinglyLinkedListNode(data);

if(head == null){
    head = newNode;
    return head;
}

if(head.next == null){
    head.next = newNode;
    newNode.next = null;
    return head;
}

SinglyLinkedListNode temp = head;

while(temp!=null){
   
    if(temp.next == null){
        temp.next = newNode;
        newNode.next = null;
    }
     temp= temp.next;
}

return head;

    }
